<?php
session_start();
 
if (!isset($_SESSION['students'])) {
  $_SESSION['students'] = [];
}
 
// Add new student at the bottom
if (isset($_POST['submit'])) {
  $student = [
    'name'  => $_POST['name'],
    'email' => $_POST['email'],
    'city'  => $_POST['city']
  ];
  $_SESSION['students'][] = $student;
}
 
// Delete specific student by index then re-index
if (isset($_GET['delete'])) {
  $index = (int) $_GET['delete'];
  if (isset($_SESSION['students'][$index])) {
    unset($_SESSION['students'][$index]);
    $_SESSION['students'] = array_values($_SESSION['students']); // re-index
  }
  header("Location: " . $_SERVER["PHP_SELF"]);
  exit;
}
 
// Clear all
if (isset($_GET['clear'])) {
  $_SESSION['students'] = [];
  header("Location: " . $_SERVER["PHP_SELF"]);
  exit;
}
 
// Show table
if (!empty($_SESSION['students'])) {
  echo "<h2>Student Records</h2>";
  echo "<table border='1' cellpadding='8' cellspacing='0'>";
  echo "<tr><th>Sr No</th><th>Name</th><th>Email</th><th>City</th><th>Action</th></tr>";
 
  foreach ($_SESSION['students'] as $index => $s) {
    echo "<tr>";
    echo "<td>" . ($index + 1) . "</td>";
    echo "<td>" . $s['name']  . "</td>";
    echo "<td>" . $s['email'] . "</td>";
    echo "<td>" . $s['city']  . "</td>";
    echo "<td><a href='?delete=" . $index . "' onclick=\"return confirm('Delete this student?')\">Delete</a></td>";
    echo "</tr>";
  }
 
  echo "</table>";
  echo "<br><a href='?clear=1'>Clear All</a>";
}
?>
