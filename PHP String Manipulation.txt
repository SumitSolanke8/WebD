        <input type="text" name="text" placeholder="Enter a string" required
               value="<?php echo isset($_POST['text']) ? htmlspecialchars($_POST['text']) : ''; ?>">

        <select name="operation" id="operation">
            <option value="length">Length</option>
            <option value="upper">Uppercase</option>
            <option value="lower">Lowercase</option>
            <option value="reverse">Reverse</option>
            <option value="wordcount">Word Count</option>
            <option value="substring">Substring</option>
        </select>

        <div id="substring-fields">
            <input type="number" name="start" placeholder="Start Index"
                   value="<?php echo isset($_POST['start']) ? $_POST['start'] : ''; ?>">
            <input type="number" name="len" placeholder="Length"
                   value="<?php echo isset($_POST['len']) ? $_POST['len'] : ''; ?>">

        <button type="submit">Submit</button>

    <?php
    $selectedOp = isset($_POST['operation']) ? $_POST['operation'] : 'length';

    if ($_SERVER["REQUEST_METHOD"] == "POST") {
        $text      = $_POST['text'];
        $operation = $_POST['operation'];
        $output    = "";

        switch ($operation) {
            case "length":
                $output = "Length : <span>" . strlen($text) . "</span>";
                break;
            case "upper":
                $output = "Uppercase : <span>" . strtoupper($text) . "</span>";
                break;
            case "lower":
                $output = "Lowercase : <span>" . strtolower($text) . "</span>";
                break;
            case "reverse":
                $output = "Reversed : <span>" . strrev($text) . "</span>";
                break;
            case "wordcount":
                $output = "Word Count : <span>" . str_word_count($text) . "</span>";
                break;
            case "substring":
                $start = $_POST['start'];
                $len   = $_POST['len'];
                if ($start !== "" && $len !== "") {
                    $output = "Substring : <span>" . substr($text, $start, $len) . "</span>";
                } else {
                    $output = "Please enter Start Index and Length.";
                }
                break;
            default:
                $output = "Invalid operation.";
        }

        echo "<div class='result'>" . $output . "</div>";
    }
    ?>

<script>
    var selectedOp = "<?php echo $selectedOp; ?>";

    var subFields = document.getElementById('substring-fields');

    // Set dropdown to previously selected value
    select.value = selectedOp;

    // Show or hide substring fields
    function toggleFields() {
        subFields.style.display = (select.value === 'substring') ? 'block' : 'none';
    }

    // Run on load to restore state after POST
    toggleFields();

    // Run on change
    select.addEventListener('change', toggleFields);
