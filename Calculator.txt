  <script>
    let current = '0';
    let expr = '';
    let operator = null;
    let prevVal = null;
    let justCalc = false;
    let waitingForNext = false; 

    function add(a, b)      { return a + b; }
    function subtract(a, b) { return a - b; }
    function multiply(a, b) { return a * b; }
    function divide(a, b)   { return b === 0 ? null : a / b; }

    function compute(a, op, b) {
      switch (op) {
        case '+': return add(a, b);
        case '-': return subtract(a, b);
        case '*': return multiply(a, b);
        case '/': return divide(a, b);
        default:  return b;
      }
    }

    function updateDisplay() {
      document.getElementById('result').textContent = current;
      document.getElementById('expr').textContent   = expr;
    }

    function inputNum(n) {
      if (justCalc || waitingForNext) {
        // After operator or = is pressed, start fresh
        current = n;
        justCalc = false;
        waitingForNext = false;
      } else {
        current = current === '0' ? n : current + n;
      }
      updateDisplay();
    }

    function inputDot() {
      if (waitingForNext) {
        current = '0.';
        waitingForNext = false;
      } else if (!current.includes('.')) {
        current += '.';
      }
      updateDisplay();
    }

    function inputOp(op) {
      justCalc = false;
      const cur = parseFloat(current);

      if (prevVal !== null && operator && !waitingForNext) {
        const res = compute(prevVal, operator, cur);
        if (res === null) {
          current = 'Error'; expr = ''; prevVal = null; operator = null;
          waitingForNext = false;
          updateDisplay(); return;
        }
        prevVal = res;
        current = formatNum(res);
      } else {
        prevVal = cur;
      }

      operator = op;
      waitingForNext = true; // Tell inputNum to clear display on next digit
      expr = formatNum(prevVal) + ' ' + opSym(op);
      updateDisplay();
    }

    function calculate() {
      if (prevVal === null || !operator) return;
      const cur = parseFloat(current);
      const res = compute(prevVal, operator, cur);
      if (res === null) {
        expr    = formatNum(prevVal) + ' ' + opSym(operator) + ' ' + current + ' =';
        current = 'Error';
        prevVal = null; operator = null; waitingForNext = false;
        updateDisplay(); return;
      }
      expr    = formatNum(prevVal) + ' ' + opSym(operator) + ' ' + current + ' =';
      current = formatNum(res);
      prevVal = null; operator = null;
      justCalc = true;
      waitingForNext = false;
      updateDisplay();
    }

    function clearAll() {
      current = '0'; expr = ''; operator = null;
      prevVal = null; justCalc = false; waitingForNext = false;
      updateDisplay();
    }

    function deleteLast() {
      if (justCalc || waitingForNext) return;
      current = current.length > 1 ? current.slice(0, -1) : '0';
      updateDisplay();
    }

    function formatNum(n) { return parseFloat(n.toFixed(10)).toString(); }
    function opSym(op) {
      return { '+': '+', '-': '−', '*': '×', '/': '÷' }[op] || op;
    }
